CREATE TABLE Products (ProductID INT(11) NOT NULL AUTO_INCREMENT, ProductName VARCHAR(40) NOT NULL,
SupplierID INT(11),
CategoryID INT(11),
QuantityPerUnit VARCHAR(20),
UnitPrice DECIMAL(10,4) DEFAULT 0.0000,
UnitsInStock SMALLINT(2) DEFAULT 0,
UnitsOnOrder SMALLINT(2) DEFAULT 0,
ReorderLevel SMALLINT(2) DEFAULT 0,
Discontinued BIT(1) NOT NULL DEFAULT b'0',
PRIMARY KEY ( ProductID )
);

5.)
SELECT ProductID, ProductName from Products WHERE Discontinued = b'1';



6.)
SELECT ProductName, UnitPrice FROM Products WHERE UnitPrice IN (SELECT MAX(UnitPrice) FROM Products)
UNION
SELECT ProductName, UnitPrice FROM Products WHERE UnitPrice IN (SELECT MIN(UnitPrice) FROM Products);



7.)
SELECT ProductID, ProductName, UnitPrice FROM Products WHERE UnitPrice < 20;



8.)
SELECT ProductID, ProductName, UnitPrice FROM Products WHERE UnitPrice < 25 AND UnitPrice > 15;



9.)
SELECT ProductID, ProductName, UnitPrice FROM Products WHERE UnitPrice > (SELECT AVG(UnitPrice) FROM Products);



10.)
SELECT ProductName, UnitPrice from Products ORDER BY UnitPrice DESC LIMIT 10;



11.)
SELECT
(SELECT Count(Discontinued) FROM Products WHERE Discontinued = b'0') as CurrentProducts,
(SELECT Count(Discontinued) FROM Products WHERE Discontinued = b'1') as DiscontinuedProducts;



12.)
